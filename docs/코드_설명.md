```java
package com.comma.global.util;

import lombok.Getter;

@Getter                                              // 모든 필드에 getter 자동 생성 (Lombok)
public class ApiResponse<T> {                        // T: 응답 데이터 타입 (제네릭)

    private final boolean success;                   // 성공 여부 (true/false)
    private final T data;                            // 실제 응답 데이터
    private final String message;                    // 사용자에게 보여줄 메시지

    private ApiResponse(boolean success, T data, String message) {  // private: 정적 팩토리 메서드로만 생성
        this.success = success;
        this.data = data;
        this.message = message;
    }

    public static <T> ApiResponse<T> ok(T data, String message) {   // 성공 응답 + 데이터 + 메시지
        return new ApiResponse<>(true, data, message);
    }

    public static <T> ApiResponse<T> ok(String message) {           // 성공 응답 + 메시지만 (데이터 없을 때)
        return new ApiResponse<>(true, null, message);
    }

    public static <T> ApiResponse<T> fail(String message) {         // 실패 응답 + 메시지
        return new ApiResponse<>(false, null, message);
    }
}
```

### `global/config/JwtUtil.java`
JWT 토큰을 생성하고 검증하는 유틸리티 클래스입니다.
```java
package com.comma.global.config;

import io.jsonwebtoken.*;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.lang.NonNull;
import org.springframework.stereotype.Component;

import java.security.Key;
import java.util.Date;

@Component                                                   // Spring Bean으로 등록
@SuppressWarnings("null")                                    // JJWT 라이브러리가 @NonNull 미지원 — 정적 분석 경고 억제
public class JwtUtil {

    private final Key key;                                   // HMAC-SHA256 서명에 사용할 키 객체
    private final long accessExpiration;                     // 액세스 토큰 만료 시간 (밀리초)
    private final long refreshExpiration;                    // 리프레시 토큰 만료 시간 (밀리초)

    public JwtUtil(
            @Value("${jwt.secret}") String secret,                          // application.yml의 jwt.secret 값 주입
            @Value("${jwt.access-expiration}") long accessExpiration,       // 액세스 토큰 만료 시간 주입
            @Value("${jwt.refresh-expiration}") long refreshExpiration      // 리프레시 토큰 만료 시간 주입
    ) {
        this.key = Keys.hmacShaKeyFor(secret.getBytes());    // 시크릿 문자열로 HMAC 키 생성
        this.accessExpiration = accessExpiration;
        this.refreshExpiration = refreshExpiration;
    }

    public @NonNull String generateAccessToken(String 쉼표번호, String role) {  // 로그인 성공 시 발급하는 단기 토큰
        return Jwts.builder()
                .setSubject(쉼표번호)                                        // 토큰 주체(subject): 쉼표번호
                .claim("role", role)                                         // 커스텀 클레임: 권한 정보
                .setIssuedAt(new Date())                                     // 발급 시각
                .setExpiration(new Date(System.currentTimeMillis() + accessExpiration))  // 만료 시각
                .signWith(key, SignatureAlgorithm.HS256)                     // HMAC-SHA256으로 서명
                .compact();                                                  // 최종 JWT 문자열 생성
    }

    public @NonNull String generateRefreshToken(String 쉼표번호) {  // 액세스 토큰 갱신에 사용하는 장기 토큰
        return Jwts.builder()
                .setSubject(쉼표번호)                                        // 주체: 쉼표번호
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + refreshExpiration))
                .signWith(key, SignatureAlgorithm.HS256)
                .compact();
    }

    public @NonNull String extract쉼표번호(String token) {    // 토큰에서 쉼표번호(subject) 추출
        return parseClaims(token).getSubject();
    }

    public @NonNull String extractRole(String token) {        // 토큰에서 권한(role 클레임) 추출
        return parseClaims(token).get("role", String.class);
    }

    /** 소셜 가입 확인 전 임시 토큰 (10분 유효) — subject: providerId, claims: provider/nickname */
    public @NonNull String generatePendingToken(String provider, String providerId, String nickname) {
        return generatePendingToken(provider, providerId, nickname, null);   // email 없는 버전 위임
    }

    /** email 포함 버전 — 구글처럼 이메일을 제공하는 소셜 로그인에서 사용 */
    public @NonNull String generatePendingToken(String provider, String providerId, String nickname, String email) {
        var builder = Jwts.builder()
                .setSubject(providerId)                                       // 소셜 제공자 고유 ID
                .claim("provider", provider)                                  // 소셜 종류 (kakao/google/naver)
                .claim("nickname", nickname != null ? nickname : "")          // 소셜에서 받아온 닉네임
                .claim("pending", true)                                       // 아직 가입 미완료 표시
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + 10 * 60 * 1000));  // 10분 후 만료
        if (email != null) builder = builder.claim("email", email);           // 이메일 있으면 추가
        return builder.signWith(key, SignatureAlgorithm.HS256).compact();
    }

    /** pending 토큰에서 클레임 추출 — 유효하지 않거나 pending 아닌 경우 null 반환 */
    public Claims extractPendingClaims(String token) {
        try {
            Claims claims = parseClaims(token);
            if (!Boolean.TRUE.equals(claims.get("pending", Boolean.class))) return null;  // pending 표시 없으면 무효
            return claims;
        } catch (JwtException | IllegalArgumentException e) {
            return null;                                                       // 파싱 실패 → null 반환
        }
    }

    public boolean isTokenValid(String token) {         // 토큰이 유효한지 검사 (서명 + 만료 시간)
        try {
            parseClaims(token);
            return true;
        } catch (JwtException | IllegalArgumentException e) {
            return false;                               // 만료되거나 서명 불일치 → false
        }
    }

    private Claims parseClaims(String token) {          // 토큰 파싱 — 서명 검증 + 클레임 추출
        return Jwts.parserBuilder()
                .setSigningKey(key)                     // 서명 검증에 사용할 키 설정
                .build()
                .parseClaimsJws(token)                  // 파싱 및 서명 검증 (실패 시 예외 발생)
                .getBody();                             // 클레임(페이로드) 반환
    }
}
```

### `global/config/JwtInterceptor.java`
모든 API 요청을 가로채 JWT 인증을 처리합니다. Spring Security 없이 인증을 구현합니다.
```java
package com.comma.global.config;

import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.lang.NonNull;
import org.springframework.stereotype.Component;
import org.springframework.util.AntPathMatcher;
import org.springframework.web.servlet.HandlerInterceptor;

import java.io.IOException;

@Component                                                   // Spring Bean으로 등록
@RequiredArgsConstructor                                     // final 필드 생성자 자동 생성 (Lombok)
public class JwtInterceptor implements HandlerInterceptor {  // HandlerInterceptor: 컨트롤러 실행 전 가로채기

    private final JwtUtil jwtUtil;                           // JWT 검증 유틸 주입

    private static final AntPathMatcher pathMatcher = new AntPathMatcher();  // /api/admin/** 패턴 매칭용

    // 로그인 없이 접근 가능한 공개 경로 목록
    private static final String[] PUBLIC_PATHS = {
            "/api/auth/signup",                  // 회원가입
            "/api/auth/login",                   // 로그인
            "/api/auth/refresh",                 // 토큰 갱신
            "/api/auth/email/verify",            // 이메일 인증 링크
            "/api/auth/password/reset-request",  // 비밀번호 재설정 요청
            "/api/auth/password/reset",          // 비밀번호 재설정 실행
            "/api/auth/check/username",          // 아이디 중복 확인
            "/api/auth/check/email",             // 이메일 중복 확인
            "/api/auth/kakao/**",                // 카카오 OAuth2 전체
            "/api/auth/google/**",               // 구글 OAuth2 전체
            "/api/auth/naver/**",                // 네이버 OAuth2 전체
            "/api/places",                       // 장소 목록 (비로그인도 탐색 가능)
            "/api/places/*",                     // 장소 상세
            "/api/rest-types",                   // 휴식 유형 목록
            "/api/survey/questions",             // 설문 문항
            "/api/contents",                     // 콘텐츠 목록
            "/api/posts",                        // 커뮤니티 게시글 목록
            "/api/challenges",                   // 챌린지 목록
            // ... (생략)
    };

    @Override
    public boolean preHandle(
            @NonNull HttpServletRequest request,
            @NonNull HttpServletResponse response,
            @NonNull Object handler) throws IOException {

        if ("OPTIONS".equalsIgnoreCase(request.getMethod())) {   // CORS preflight 요청은 무조건 통과
            return true;
        }

        String path = request.getServletPath();                  // 요청 경로 추출 (예: /api/auth/login)

        // 공개 경로인지 확인 (AntPathMatcher로 와일드카드 패턴 지원)
        boolean isPublic = false;
        for (String pattern : PUBLIC_PATHS) {
            if (pathMatcher.match(pattern, path)) {
                isPublic = true;
                break;
            }
        }

        if (isPublic) {
            // 공개 경로라도 토큰이 있으면 사용자 정보를 추출 (선택적 인증 — 리뷰 작성자 표시 등)
            String authHeader = request.getHeader("Authorization");
            if (authHeader != null && authHeader.startsWith("Bearer ")) {
                String token = authHeader.substring(7);          // "Bearer " 이후 토큰 문자열만 추출
                if (jwtUtil.isTokenValid(token)) {
                    request.setAttribute("쉼표번호", jwtUtil.extract쉼표번호(token));  // 컨트롤러에서 사용 가능하도록 저장
                    request.setAttribute("role", jwtUtil.extractRole(token));
                }
            }
            return true;                                         // 공개 경로 → 통과
        }

        // 비공개 경로: Authorization 헤더 확인
        String header = request.getHeader("Authorization");
        if (header == null || !header.startsWith("Bearer ")) {
            sendError(response, 401, "로그인이 필요합니다.");     // 헤더 없으면 401 반환
            return false;
        }

        String token = header.substring(7);                      // "Bearer " 이후 토큰 추출

        if (!jwtUtil.isTokenValid(token)) {
            sendError(response, 401, "유효하지 않은 토큰입니다. 다시 로그인해주세요.");  // 만료/위조 토큰
            return false;
        }

        String 쉼표번호 = jwtUtil.extract쉼표번호(token);
        String role = jwtUtil.extractRole(token);

        // /api/admin/** 경로는 ADMIN 권한만 허용
        if (pathMatcher.match("/api/admin/**", path) && !"ADMIN".equals(role)) {
            sendError(response, 403, "관리자 권한이 필요합니다.");
            return false;
        }

        request.setAttribute("쉼표번호", 쉼표번호);              // 컨트롤러에서 request.getAttribute("쉼표번호")로 접근
        request.setAttribute("role", role);

        return true;                                             // 인증 성공 → 컨트롤러로 전달
    }

    private void sendError(HttpServletResponse response, int status, String message) throws IOException {
        response.setStatus(status);                              // HTTP 상태 코드 설정
        response.setContentType("application/json;charset=UTF-8");
        response.getWriter().write(                              // 응답 본문에 JSON 직접 작성
                String.format("{\"success\":false,\"data\":null,\"message\":\"%s\"}", message)
        );
    }
}
```

### `global/config/CorsConfig.java`
프론트엔드(React)가 백엔드 API를 호출할 수 있도록 CORS(교차 출처 리소스 공유)를 설정합니다.

```java
package com.comma.global.config;

import org.springframework.context.annotation.Configuration;
import org.springframework.lang.NonNull;
import org.springframework.web.servlet.config.annotation.CorsRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@Configuration                                                   // 설정 클래스임을 표시
public class CorsConfig implements WebMvcConfigurer {            // WebMvc 설정 커스터마이징

    @Override
    public void addCorsMappings(@NonNull CorsRegistry registry) {
        registry.addMapping("/api/**")                           // /api/ 하위 모든 경로에 CORS 적용
                .allowedOrigins(
                    "http://localhost:3000",                     // React 개발 서버 (Create React App)
                    "http://localhost:5173"                      // Vite 개발 서버 (대안)
                )
                .allowedMethods("GET", "POST", "PUT", "DELETE", "OPTIONS", "PATCH")  // 허용 HTTP 메서드
                .allowedHeaders("*")                             // 모든 요청 헤더 허용
                .exposedHeaders("Authorization")                 // 응답 헤더 중 Authorization을 JS에서 읽기 허용
                .allowCredentials(true)                          // 쿠키/인증 정보 포함 요청 허용
                .maxAge(3600);                                   // preflight 캐시 시간 (1시간)
    }
}
```

### `global/config/WebConfig.java`
인터셉터를 등록하고, 업로드 파일을 정적 리소스로 서빙합니다.
```java
package com.comma.global.config;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Configuration;
import org.springframework.lang.NonNull;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

import java.nio.file.Paths;

@Configuration
public class WebConfig implements WebMvcConfigurer {

    private final @NonNull JwtInterceptor jwtInterceptor;   // JWT 인증 인터셉터 주입

    @Value("${app.upload.dir:uploads}")                     // application.yml의 업로드 디렉토리 경로
    private String uploadDir;

    public WebConfig(@NonNull JwtInterceptor jwtInterceptor) {
        this.jwtInterceptor = jwtInterceptor;
    }

    @Override
    public void addInterceptors(@NonNull InterceptorRegistry registry) {
        registry.addInterceptor(jwtInterceptor)             // JwtInterceptor를 인터셉터로 등록
                .addPathPatterns("/api/**");                 // /api/ 하위 모든 요청에만 적용
    }

    // 업로드 파일을 정적 URL로 서빙 — /uploads/xxx.jpg → 서버 로컬 파일 시스템의 파일 반환
    @Override
    public void addResourceHandlers(@NonNull ResourceHandlerRegistry registry) {
        String absolutePath = Paths.get(uploadDir).toAbsolutePath().toString();  // 상대경로 → 절대경로 변환
        registry.addResourceHandler("/uploads/**")                               // 이 URL 패턴으로 요청이 오면
                .addResourceLocations("file:" + absolutePath + "/");             // 이 디렉토리에서 파일 찾아 반환
    }
}
```

### `global/util/AnalyticsService.java`
사용자 행동 이벤트를 비동기로 DB에 저장합니다. 실패해도 서비스에 영향이 없습니다.
```java
package com.comma.global.util;

import com.comma.domain.admin.mapper.AdminMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;

import java.util.Map;

@Slf4j                                              // log 객체 자동 생성 (Lombok)
@Service
@RequiredArgsConstructor
public class AnalyticsService {

    private final AdminMapper adminMapper;           // analytics_events 테이블에 저장하는 매퍼

    @Async                                           // 이 메서드는 별도 스레드에서 실행 — 호출한 쪽은 바로 반환
    public void track(String commaNo, String eventType, Map<String, Object> data) {
        try {
            String json = data == null ? null : toJson(data);  // Map → JSON 문자열 변환
            adminMapper.insertAnalyticsEvent(commaNo, eventType, json);  // DB 저장
        } catch (Exception e) {
            log.warn("analytics track 실패: {} - {}", eventType, e.getMessage());  // 실패해도 경고만 출력
        }
    }

    // 외부 Jackson 없이 간단한 JSON 직렬화 (의존성 최소화)
    private String toJson(Map<String, Object> data) {
        StringBuilder sb = new StringBuilder("{");
        boolean first = true;
        for (Map.Entry<String, Object> entry : data.entrySet()) {
            if (!first) sb.append(",");                         // 첫 항목 이후에만 쉼표 추가
            sb.append("\"").append(entry.getKey()).append("\":");  // "키":
            Object val = entry.getValue();
            if (val == null) {
                sb.append("null");
            } else if (val instanceof Number) {
                sb.append(val);                                 // 숫자는 따옴표 없이
            } else if (val instanceof Boolean) {
                sb.append(val);                                 // boolean도 따옴표 없이
            } else {
                sb.append("\"").append(val.toString().replace("\"", "\\\"")).append("\"");  // 문자열은 이스케이프
            }
            first = false;
        }
        sb.append("}");
        return sb.toString();
    }
}
```

### `domain/auth/service/AuthService.java` (회원가입 부분)

```java
@Transactional                                                      // 가입 중 오류 시 모든 DB 변경 롤백
public User signup(String email, String username, String password) {
    // 이미 사용 중인 이메일/아이디 체크
    if (authMapper.countByEmail(email.trim()) > 0) {
        throw new IllegalArgumentException("이미 사용 중인 이메일입니다.");
    }
    if (authMapper.countByUsername(username.trim()) > 0) {
        throw new IllegalArgumentException("이미 사용 중인 아이디입니다.");
    }

    // 동시 가입 시 race condition 방지 — MAX 조회 후 즉시 INSERT (같은 트랜잭션)
    String 쉼표번호 = generate쉼표번호();                              // DB MAX(쉼표번호) + 1 방식으로 순번 생성

    User user = new User();
    user.set쉼표번호(쉼표번호);
    user.setEmail(email.trim());
    user.setPassword(BCrypt.hashpw(password, BCrypt.gensalt()));     // BCrypt로 비밀번호 단방향 암호화
    user.setUsername(username.trim());
    user.setNickname(쉼표번호);                                       // 닉네임 초기값 = 쉼표번호 (나중에 변경 가능)
    user.setStatus("active");                                         // 가입 즉시 활성 상태
    user.setEmailVerified(false);                                     // 이메일 인증 전
    user.setRole("USER");

    authMapper.insertUser(user);

    if (skipEmailVerification) {                                      // 로컬 개발 환경: 이메일 인증 자동 완료 처리
        authMapper.updateUserEmailVerified(쉼표번호);
        user.setEmailVerified(true);
    } else {                                                          // 운영 환경: 인증 메일 발송
        try {
            String token = UUID.randomUUID().toString();              // 무작위 UUID를 인증 토큰으로 사용
            authMapper.insertEmailVerification(쉼표번호, token, LocalDateTime.now().plusHours(24));  // 24시간 유효
            mailService.sendVerificationEmail(email.trim(), username.trim(), token);
        } catch (Exception e) {
            // 메일 발송 실패해도 회원가입은 성공 처리 (사용자가 이미 가입됨)
            log.error("[메일 발송 실패] to={}, error={}", email.trim(), e.getMessage(), e);
        }
    }
    // ...
}
```

### `api/fetchWithAuth.js`
JWT 인증이 필요한 API를 호출하는 래퍼 함수입니다. 토큰 만료 시 자동 갱신합니다.
```js
/**
 * 인증이 필요한 API 호출 유틸
 * - Authorization 헤더 자동 추가
 * - 401 응답 시 refreshToken으로 accessToken 갱신 후 재시도
 * - 갱신 실패 시 로그아웃 처리 (localStorage 클리어 + /login 리다이렉트)
 */

let isRefreshing = false;    // 현재 토큰 갱신 중인지 여부 (전역 플래그)
let refreshQueue = [];       // 갱신 중에 들어온 다른 요청들을 대기시키는 큐

// 갱신 완료 후 큐에 대기 중인 요청들에게 새 토큰 전달
function processRefreshQueue(newToken) {
  refreshQueue.forEach(({ resolve }) => resolve(newToken));
  refreshQueue = [];
}

// 갱신 실패 시 큐에 대기 중인 요청들에게 에러 전달
function rejectRefreshQueue(error) {
  refreshQueue.forEach(({ reject }) => reject(error));
  refreshQueue = [];
}

// 토큰 삭제 + 로그인 페이지 이동 (강제 로그아웃)
function forceLogout() {
  localStorage.removeItem('accessToken');
  localStorage.removeItem('refreshToken');
  localStorage.removeItem('user');
  window.location.href = '/login';
}

// Refresh Token으로 새 Access Token 발급 요청
async function refreshAccessToken() {
  const refreshToken = localStorage.getItem('refreshToken');
  if (!refreshToken) {
    throw new Error('리프레시 토큰 없음');   // Refresh Token 자체가 없으면 재로그인 필요
  }

  const res = await fetch('/api/auth/refresh', {  // 토큰 갱신 API 호출
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ refreshToken })
  });
  const data = await res.json();

  if (!data.success) {
    throw new Error(data.message);            // 갱신 실패 (Refresh Token도 만료된 경우)
  }

  localStorage.setItem('accessToken', data.data.accessToken);  // 새 Access Token 저장
  return data.data.accessToken;
}

/**
 * fetch wrapper with automatic token refresh
 * @param {string} url - API URL (예: '/api/auth/me')
 * @param {object} options - fetch options (method, body 등)
 * @returns {Promise<object>} parsed JSON response
 */
export async function fetchWithAuth(url, options = {}) {
  const accessToken = localStorage.getItem('accessToken');  // 현재 저장된 Access Token

  const headers = {
    'Content-Type': 'application/json',    // 기본 JSON 타입 헤더
    ...options.headers,                    // 호출 측에서 추가한 헤더로 덮어쓰기 가능
  };

  if (accessToken) {
    headers['Authorization'] = `Bearer ${accessToken}`;   // Access Token을 Bearer 방식으로 전달
  }

  let res = await fetch(url, { ...options, headers });    // 실제 API 호출

  // 401 응답 = 토큰 만료 → 갱신 시도
  if (res.status === 401 && localStorage.getItem('refreshToken')) {
    try {
      let newToken;

      if (isRefreshing) {
        // 이미 다른 요청이 갱신 중이면 큐에 등록하고 갱신 완료까지 대기
        newToken = await new Promise((resolve, reject) => {
          refreshQueue.push({ resolve, reject });
        });
      } else {
        isRefreshing = true;                               // 갱신 시작 플래그
        try {
          newToken = await refreshAccessToken();           // 새 Access Token 발급
          processRefreshQueue(newToken);                   // 대기 중인 요청들에게 새 토큰 전달
        } catch (refreshError) {
          rejectRefreshQueue(refreshError);                // 갱신 실패 → 대기 요청들에게 에러 전달
          forceLogout();                                   // 강제 로그아웃
          throw refreshError;
        } finally {
          isRefreshing = false;                            // 갱신 완료 플래그 해제
        }
      }

      // 새 토큰으로 원래 요청 재시도
      headers['Authorization'] = `Bearer ${newToken}`;
      res = await fetch(url, { ...options, headers });
    } catch (e) {
      forceLogout();
      throw e;
    }
  }

  return res.json();   // 최종 응답을 JSON으로 파싱해 반환
}

export default fetchWithAuth;
```

### `api/analytics.js`
사용자 행동 이벤트를 백엔드로 전송합니다. 실패해도 조용히 무시합니다.
```js
/**
 * 사용자 행동 이벤트 수집 유틸
 * - 실패해도 조용히 무시 (서비스 영향 없음)
 */
export async function trackEvent(eventType, data = {}) {
  try {
    const token = localStorage.getItem('accessToken');   // 로그인 토큰 (없으면 비로그인 이벤트로 수집)
    await fetch('/api/analytics/track', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...(token ? { Authorization: `Bearer ${token}` } : {}),  // 토큰 있을 때만 Authorization 헤더 추가
      },
      body: JSON.stringify({ eventType, data }),         // 이벤트 유형과 부가 데이터 전송
    });
  } catch {
    // 네트워크 오류 등 실패 → 완전히 무시 (콘솔 출력도 안 함)
  }
}

// 페이지 진입 이벤트 전용 단축 함수
export function trackPageView(pageName) {

}
```
